# Install

```bash
npm install --save rimble-ui styled-components
```

## Usage

```jsx
import React, { Component } from 'react'
```

```jsx
import { Button } from 'rimble-ui'
```

```jsx
class Example extends Component {
  render () {
    return (
      <Button size={'medium'}>
        Click me!
      </Button>
    )
  }
}
```
